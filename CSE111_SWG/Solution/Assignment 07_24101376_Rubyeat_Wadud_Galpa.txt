public class Shape2D {
  public int length;
  public int breadth;
  public int height;
  public int side1, side2, side3;
  public double area;
  public String shape;

  public Shape2D(int l) {
    length = l;
    shape = "Square";
    System.out.println("A Square has been created with length: " + l);
    area = l * l;
  }
  public Shape2D(int l, int b) {
    length = l;
    breadth = b;
    shape = "Rectangle";
    System.out.println("A Rectangle has been created with length: " + l + " and breadth: " + b);
    area = l * b;
  }
  public Shape2D(int h, int b, String s) {
    height = h;
    breadth = b;
    shape = "Triangle";
    System.out.println("A Triangle has been created with height: " + h + " and base: " + b);
    area = 0.5 * h * b;
  }
  public Shape2D(int a, int b, int c) {
    side1 = a;
    side2 = b;
    side3 = c;
    shape = "Triangle";
    System.out.println("A Triangle has been created with the following sides: " + side1 + ", " + side2 + ", " + side3);
    int s = (a+b+c)/2;
    area = Math.sqrt(s * (s - side1) * (s - side2) * (s - side3));;
  }
  public void area() {
    System.out.println("The area of the "+ shape +" is: " + area);
  }
}

public class Student {
  private int id;
  private double cgpa;
  private String[] courses;
  private int count;
  
  public Student(int x) {
    id = x;
    cgpa = 0.0; 
    courses = new String[4];
    count = 0;
  }
  public Student(int x, double y) {
    id = x;
    cgpa = y;
    courses = new String[4];
    count = 0;
  }
  public void setCG(double y) {
    cgpa = y;
  }
  public void setID(int x) {
    id = x;
  }
  public void addCourse(String course) {
    if (cgpa == 0.0) {
      System.out.println("Failed to add " + course);
      System.out.println("Set CG first");
    } else if (cgpa < 3.0 && count >= 3) {
      System.out.println("Failed to add " + course);
      System.out.println("CG is low. Can't add more than 3 courses.");
    } else if (count >= 4) {
      System.out.println("Failed to add " + course);
      System.out.println("Maximum 4 courses allowed.");
    } else {
      courses[count] = course;
      count++;
    }
  }
  public void addCourse(String[] courseArr) {
    for (int i = 0; i < courseArr.length; i++) {
      addCourse(courseArr[i]);
    }
  }
  public void rmAllCourse() {
    for (int i = 0; i < count; i++) {
      courses[i] = null; 
    }
    count = 0;
  }
  public void showAdvisee() {
    System.out.println("Student ID: " + id + ", CGPA: " + cgpa);
    if (count == 0) {
      System.out.println("No courses added.");
    } else {
      System.out.print("Added courses are:\n");
      for (int i = 0; i < count; i++) {
        System.out.print(courses[i] + " ");
      }
      System.out.println();
    }
  }
}

public class ABCServer {
  private String name;
  private int capacity;
  private String[] members;
  private String[] roles;
  private int totalMembers;
  
  public ABCServer() {
    name = "Default";
    capacity = 10;
    members = new String[capacity];
    roles = new String[capacity];
    totalMembers = 0;
  }
  public ABCServer(String n, int cap) {
    name = n;
    capacity = cap;
    members = new String[capacity];
    roles = new String[capacity];
    totalMembers = 0;
  }
  public void addMembers(String member) {
    if (totalMembers < capacity) {
      members[totalMembers] = member;
      roles[totalMembers] = "Rising Hero";
      totalMembers++;
      System.out.println("Rising Hero is added.");
    } else {
      System.out.println("Sorry, maximum capacity exceeded!");
    }
  }
  public void addMembers(String member, String role) {
    if (totalMembers < capacity) {
      members[totalMembers] = member;
      roles[totalMembers] = role;
      totalMembers++;
      System.out.println(role + " is added.");
    } else {
      System.out.println("Sorry, maximum capacity exceeded!");
    }
  }
  public void details() {
    System.out.println("Server Name: " + name);
    System.out.println("Member Capacity: " + capacity);
    System.out.println("Total Members: " + totalMembers);
    System.out.println("Members:");
    for (int i = 0; i < totalMembers; i++) {
      System.out.println("Name:Role --> " + members[i] + ":" + roles[i]);
    }
  }
}
