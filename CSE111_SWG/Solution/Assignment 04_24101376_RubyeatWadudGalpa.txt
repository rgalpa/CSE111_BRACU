//Task8
public class Bird {
  String name;
  int height = 0;
  public void makeNoise() {
    if("Parrot".equals(this.name)) {
      System.out.println("Squawk");
    }else if ("Eagle".equals(this.name)) {
      System.out.println("Squee");
    }
  }
  public void flyUp(int feet) {
    this.height += feet;
    System.out.println(this.name + " has flown up " + feet + " feet.");
  }
  public void flyDown(int feet) {
    if (this.height < feet) {
      System.out.println(this.name + " cannot fly down " + feet + " feet.");
    } else {
      this.height -= feet;
      if (this.height == 0) {
        System.out.println(this.name + " has flown down " + feet + " feet and landed.");
      } else {
        System.out.println(this.name + " has flown down " + feet + " feet.");
      }
    }
  }
}


//Task9
public class CellPhone {
  String model = "unknown";
  String[] contacts = new String[3];
  int Count = 0;
  public void storeContact(String name) {
    if (this.Count < this.contacts.length) {
      this.contacts[this.Count] = name;
      this.Count++;
      System.out.println("Contact Stored");
    } else {
      System.out.println("Memory full. New name can't be stored.");
    }
  }
  public void printDetails() {
    System.out.println("Phone Model " + model);
    System.out.println("Contacts Stored " + this.Count);
    if (this.Count > 0) {
      System.out.println("Stored Contacts:");
      for (int i = 0; i < this.Count; i++) {
        System.out.println(this.contacts[i]);
      }
    }
  }
}
